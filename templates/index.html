@app.route("/", methods=["GET", "POST"]) def index(): response = "" if request.method == "POST": try: prompt = request.form["prompt"] if prompt.strip() == "": response = "⚠️ กรุณาพิมพ์คำถามก่อนกดส่ง" else: chat = openai.ChatCompletion.create( model="gpt-3.5-turbo", messages=[{"role": "user", "content": prompt}] ) response = chat.choices[0].message["content"] except Exception as e: response = f"❌ ERROR: {str(e)}" return render_template("index.html", response=response)
